    function loadDocument(docId) {
        const doc = documents[docId];
        const viewer = document.getElementById('document-viewer');
        
        if (doc) {
            // Convert plain text with markdown-style formatting to HTML
            let htmlContent = doc.content
                .replace(/^# (.*$)/gm, '<h1>$1</h1>')
                .replace(/^## (.*$)/gm, '<h2>$1</h2>')
                .replace(/^### (.*$)/gm, '<h3>$1</h3>')
                .replace(/^#### (.*$)/gm, '<h4>$1</h4>')
                .replace(/\*\*(.*?)\*\*/g, '<strong>$1</strong>')
                .replace(/\n\n/g, '</p><p>')
                .replace(/\n/g, '<br>')
                .replace(/^- (.*$)/gm, '<li>$1</li>')
                .replace(/```([^`]+)```/g, '<pre><code>$1</code></pre>');
            
            // Wrap lists
            htmlContent = htmlContent.replace(/(<li>.*<\/li>)\n(?!<li>)/g, '$1</ul>');
            htmlContent = htmlContent.replace(/(<li>.*?)(?=<h|<p|$)/g, '<ul>$1</ul>');
            
            viewer.innerHTML = `
                <div style="background: rgba(255,255,255,0.05); border-radius: 10px; padding: 2rem;">
                    <h1 class="text-gradient" style="margin-bottom: 1rem;">${doc.title}</h1>
                    <div style="color: #e3f2fd; line-height: 1.8;">
                        ${htmlContent}
                    </div>
                </div>
            `;
        }
        
        // Update URL hash without reloading
        window.location.hash = docId;
    }

    // Check if there's a hash in the URL on page load
    window.onload = function() {
        if (window.location.hash) {
            const docId = window.location.hash.substring(1);
            loadDocument(docId);
        }
    };
